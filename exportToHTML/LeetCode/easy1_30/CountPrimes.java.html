package LeetCode.easy1_30;

public class CountPrimes {

    public static int countPrimes(int n) {

        if(n==0 || n== 1){
            return 0;
        }

        int count =0;
        int i=2;
        while(i<=n){

            if(i%2 == 0){
                System.out.println(i);
                count++;
            }
            i++;
        }


        return count;

    }



    public static void main(String[]args){

        System.out.println(countPrimes(10));

    }
}
